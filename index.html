# Import and initialize
from herringbone import DecisionAPI
api = DecisionAPI('your-api-key')

# Analyze any text
result = api.analyze.quick(
    text="Should we expand to EU market?"
)

# Get structured output
print(result.decision)     # "Delay expansion"
print(result.confidence)   # 0.87
print(result.reasoning)    # {...}
print(result.alternatives) # [...]
